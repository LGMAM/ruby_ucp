70:   def checksum(s)
71:     # The <checksum> is derived by the addition of all bytes of the header, data field separators
72:     # and data fields (i.e. all characters after the stx-character, up to and including the last “/”
73:     # before the checksum field). The 8 Least Significant Bits (LSB) of the result is then
74:     # represented as two printable characters.
75: 
76:     sum=0
77:     s.each_byte { |byte|
78:       sum+=byte
79:     }
80:     #return sprintf("%02x",sum)[-2,2]
81:     return sum.to_s(16)[-2,2].upcase
82:   end
