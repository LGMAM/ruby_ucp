  def self.packoadc(oa)
    packedoa=UCP.pack7bits(oa)

    # esta conta nao esta correcta... por causa das extensoes...
    #useful_nibbles=packedoa.length
    useful_nibbles=(oa.length*7.0/4).ceil

    tmp=useful_nibbles.to_s(16).upcase
    if tmp.length==1
      tmp="0"+tmp
    end

    return tmp+packedoa
  end
