  def self.initialize_ascii2ira
    
    ('A'..'Z').each { |c|
      @gsmtable[c]=c[0]
      @asciitable[c[0]]=c
    }
    ('a'..'z').each { |c|
      @gsmtable[c]=c[0]
      @asciitable[c[0]]=c
    }
    ('0'..'9').each { |c|
      @gsmtable[c]=c[0]
      @asciitable[c[0]]=c
    }

    @gsmtable["@"]=0
    @asciitable[0]="@"

    @gsmtable[" "]=32


    @extensiontable["€"]=0x65
    @extensiontable_rev[0x65]="€"



  end
