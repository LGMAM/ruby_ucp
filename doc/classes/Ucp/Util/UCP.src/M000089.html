  def self.decode_ira(str)
    unencoded=""

    i=0
    while i<str.length
      hexv=str[i,2]
      if "1B".eql?(hexv)
        i+=2
        hexv=str[i,2]

        value=@extensiontable_rev[hexv.to_i(16)]
        value=" " if value.nil?
        unencoded+=value
      else
        val=@asciitable[hexv.to_i(16)]
        val=" " if val.nil?
        unencoded+=val
      end

      i+=2
    end

    return unencoded
  end
