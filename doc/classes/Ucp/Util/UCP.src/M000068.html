def self.str2ucs2(str,max_bytes=nil)
  hexstr=""
  str=Iconv.iconv("utf-16be", "utf-8", str).first
  i=0
  str.each_byte{ |c|
    hexstr+=UCP.int2hex(c)
    i+=1
    if (!max_bytes.nil?) && (i == max_bytes)
      break
    end
  }

  return Ucs2PackedMsg.new(hexstr,UCP.utf8_substr(str,0,i-1),i/2,i)
end
