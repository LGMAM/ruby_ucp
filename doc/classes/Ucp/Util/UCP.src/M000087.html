  def self.hextobin_reversed(hstr)
    bstr=""
    i=0
    while i<hstr.length
      tmp=hstr[i,2].to_i(16).to_s(2)

      nfillbits=8-tmp.length
      if nfillbits!=0
        tmp="0"*nfillbits+tmp
      end

      bstr=tmp+bstr
      i+=2
    end
    return bstr
  end
