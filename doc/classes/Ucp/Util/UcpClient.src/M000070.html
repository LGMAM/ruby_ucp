  def connect
    puts "connect()"
    
    begin
        @socket = TCPSocket.new(@host, @port)
        @connected=true
    rescue
        @connected=false
        return false
    end


    if !@authcreds.nil?
      auth_ucp=Ucp60Operation.new
      auth_ucp.basic_auth(@authcreds[:login],@authcreds[:password])
      auth_ucp.trn="00"
      answer=send_sync(auth_ucp)
      #puts "Crecv: #{answer.to_s}\n"
      
      if !answer.nil? && answer.is_ack?
        inc_trn()
        return true
      else
        close()
        return false
      end
    end

  end
