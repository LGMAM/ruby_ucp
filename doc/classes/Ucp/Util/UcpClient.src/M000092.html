  def send_alert(recipient,pid)
    ucp=Ucp31Operation.new
    ucp.basic_alert(recipient,pid)

    trn=UCP.int2hex(@trn)
    ucp.trn=trn
    ans=send_sync(ucp)
    inc_trn()
#      puts "e um ack: #{ans.nil?} ; #{ans.is_ack?}"
    return false if ans.nil? || !ans.is_ack?

    return true
  end
