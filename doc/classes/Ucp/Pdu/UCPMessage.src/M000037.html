  def to_s

    s=""
    @fields.each{ |key|
      value=@h[key]
      if value.nil?
        s+=DELIMITER
      else
#        if key.equal?(:msg)
#          value=value.to_s.unpack("H*").to_s.upcase!
#        end

        s+=value.to_s+DELIMITER
      end
    }


    pdu=@trn+DELIMITER
    pdu+=length(s)+DELIMITER
    pdu+=@operation_type+DELIMITER
    pdu+=@operation+DELIMITER
    pdu+=s
    pdu+=checksum(pdu)
    pdu=STX+pdu+ETX
    return pdu
  end
