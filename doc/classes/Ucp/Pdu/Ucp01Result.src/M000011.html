  def initialize(fields=nil)
    super()
    @operation_type="R"
    if fields.nil?
      return
    end

    @trn=fields[0]
    if fields[4].eql?("A")
      # 06/00043/R/01/A/01234567890:090196103258/4E
      ack(fields[5])
    elsif fields[4].eql?("N")
      # 12/00022/R/01/N/02//03
      nack(fields[5],fields[6])
    else
      raise "invalid result in UCP01"
    end

    # TODO: verificar len e checksum

  end
