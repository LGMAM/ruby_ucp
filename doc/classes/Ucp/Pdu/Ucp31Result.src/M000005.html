  def initialize(fields=nil)
    super()
    @operation_type="R"
    if fields.nil?
      return
    end

    @trn=fields[0]
    if fields[4].eql?("A")
      # 04/00024/R/31/A/0003/5D
      ack(fields[5])
    elsif fields[4].eql?("N")
      # 00/00022/R/31/N/06//07
      nack(fields[5],fields[6])
    else
      raise "invalid result in UCP31"
    end

    # TODO: verificar len e checksum

  end
