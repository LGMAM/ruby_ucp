  def initialize(fields=nil)
    super()
    @operation_type="R"
    if fields.nil?
      return
    end

    @trn=fields[0]
    if fields[4].eql?("A")
      # 00/00019/R/60/A//6D
      ack(fields[5])
    elsif fields[4].eql?("N")
      # 00/00022/R/60/N/01//04
      nack(fields[5],fields[6])
    else
      raise "invalid result in UCP60"
    end

    # TODO: verificar len e checksum
    
  end
