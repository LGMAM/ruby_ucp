  def to_sx
    
    s=""
    #puts @fields.join("/")
    #return "ola"+@h["oadc"].to_s
    @fields.each{ |key|
      value=@h[key]
      if value.nil?
        s+=DELIMITER+DELIMITER
      else
        if key=="msg"
          value=value.to_s.unpack("H*").to_s.upcase!
        end

        s+=value.to_s+DELIMITER
      end
    }
    

    super(s)
  end
